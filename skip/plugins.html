# 截图插件

# 安装

 npm install vue-web-screen-shot

# 引入

  // 导入截屏插件
  import screenShort from "vue-web-screen-shot";
  const app = createApp(App);
  // 使用截屏插件
  app.use(screenShort, { enableWebRtc: false })
